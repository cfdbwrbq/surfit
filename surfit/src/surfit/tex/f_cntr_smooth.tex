Problem: we need to approximate isoline with cell values. To do
this we add specific condition to system of linear equations:
\begin{equation}\label{phi_0}
\Phi = \sum\limits_{i=1}^N \left( f(x_i) - y_i \right)^2
\rightarrow \min,
\end{equation}
where $x_i$ - are coordinates where isoline cross sect between two
grid nodes, $y_i$ - isoline value,
\begin{equation}\label{f_x}
 f(x) = u_0 + (u_1-u_0)
\frac{x-\overline{x}_0}{\overline{x}_1-\overline{x}_0}.
\end{equation}
With respect to \eqref{f_x} we can rewrite \eqref{phi_0}:
\begin{equation}\label{phi}
\Phi = \sum\limits_{i=1}^N \left( u_0 + (u_1-u_0)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} - y_i
\right)^2 \rightarrow \min.
\end{equation}
To find minimum of this functional we should solve the system of
linear equations:
$$
\frac{ \partial \Phi } { \partial u_0 } =
%
2 \sum_i \left( u_0 + (u_1-u_0)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} - y_i
\right)
%
\left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right) =
$$
$$
= 2 \sum_i \left( \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
u_0 + \frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} u_1
- y_i \right) \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right) =
$$
$$
= 2 u_0 \sum_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)^2
%
+ 2 u_1 \sum_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}
%
- 2 \sum_i y_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
=0
$$

%
%
%
%
%

$$
\frac{ \partial \Phi } { \partial u_1 } =
%
2 \sum_i \left( u_0 + (u_1-u_0)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} - y_i
\right)
%
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} =
$$
$$
= 2 \sum_i \left( \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
u_0 + \frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} u_1
- y_i \right)
%
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} =
$$
$$
= 2 u_0 \sum_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}
%
+ 2 u_1 \left(
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)^2
%
- 2 \sum_i y_i
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} = 0
$$

$$
\begin{cases}
u_0 \underbrace{\sum\limits_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}
\right)^2}_{sum1}
%
+ \, u_1 \underbrace{\sum\limits_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}}_{sum2}
%
= \, \underbrace{\sum\limits_i y_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}
\right)}_{sum4}
\\
u_0 \underbrace{\sum\limits_i \left( 1 -
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0} \right)
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}}_{sum2}
%
+ u_1 \, \underbrace{\sum\limits_i \left(
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}
\right)^2}_{sum3}
%
= \, \underbrace{\sum\limits_i y_i
\frac{x_i-\overline{x}_0}{\overline{x}_1-\overline{x}_0}}_{sum5}
\end{cases}
$$
